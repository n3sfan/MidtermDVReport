\subsubsection{Biểu đồ nến SOL/USD}
\begin{figure}[H]
    \centering

    \caption{Biểu đồ nến SOL/USD cùng Khối lượng giao dịch và Đường Trung bình 7 \& 25}

\textbf{Giải thích thiết kế:}
Dựa trên lý thuyết trực quan hóa dữ liệu, biểu đồ này sử dụng các dấu hiệu và kênh hiển thị như sau để mã hóa thông tin:

\begin{itemize}
    \item \textbf{Thành phần nến giá:}
    \begin{itemize}
        \item \textbf{Dấu hiệu:} Hình chữ nhật thân nến - Giá mở/Giá đóng và Đường thẳng bóng nến - Cao nhất/Thấp nhất.
        \item \textbf{Kênh - Vị trí:} Trục X là thời gian, Trục Y là mức giá.
        \item \textbf{Kênh - Kích thước:} Chiều dài thân nến thể hiện độ chênh lệch giữa giá mở cửa và đóng cửa, trong khi thân bóng nến biểu thị biên độ biến động giá trong phiên.
        \item \textbf{Kênh - Màu sắc:} Dùng để mã hóa tính chất thay đổi. Màu xanh cho nến đóng cao hơn giá mở, và màu đỏ/hồng cho nến đóng thấp hơn giá mở.
    \end{itemize}
    
    \item \textbf{Thành phần đường trung bình động (MA7, MA25):}
    \begin{itemize}
        \item \textbf{Dấu hiệu:} Đường cong liên tục.
        \item \textbf{Kênh - Vị trí:} Trục Y thể hiện mức giá trung bình ứng trên trục X thời gian.
        \item \textbf{Kênh - Màu sắc:} Sử dụng màu cam cho MA(7) và màu tím cho MA(25). Mục đích là để phân biệt hai chuỗi dữ liệu đường trung bình khác nhau.
    \end{itemize}

    \item \textbf{Thành phần khối lượng giao dịch:}
    \begin{itemize}
        \item \textbf{Dấu hiệu:} Cột dọc hẹp.
        \item \textbf{Kênh - Vị trí:} Đồng bộ trục X thời gian với nến giá phía trên, nhưng trục Y được đặt độc lập thấp bên dưới với thang đo riêng về khối lượng.
        \item \textbf{Kênh - Kích thước:} Chiều cao của cột trên trục Y thể hiện độ lớn định lượng khối lượng giao dịch.
        \item \textbf{Kênh - Màu sắc:} Đồng bộ sắc độ với màu của nến trên cùng khung giờ (xanh hoặc đỏ) để người xem có thể đánh giá nhanh tương quan khối lượng với việc giá tăng/giảm.
    \end{itemize}
\end{itemize}

\textbf{Nhận xét:}
Sự kết hợp giữa Nến giá, MA và Volume giúp người dùng xác định không chỉ xu hướng giá mà còn cả sức mạnh của xu hướng đó thông qua khối lượng đi kèm.

\textbf{Giải thích thiết kế:}
Biểu đồ này giải quyết nhiệm vụ phân tích sự cam kết của thị trường đối với một xu hướng giá.
\begin{itemize}
    \item \textbf{Mã hóa thị giác:} Sử dụng hai trục Y để hiển thị đồng thời hai biến định lượng có thang đo khác nhau trên cùng một trục thời gian.
    \item \textbf{Kênh - Vị trí:} Trục X là thời gian, giúp theo dõi sự thay đổi liên tục.
    \item \textbf{Kênh - Màu sắc:} Sử dụng màu vàng cho đường giá để tạo sự tương phản mạnh trên nền tối, và màu xanh dương/tím cho vùng Open Interest.
    \item \textbf{Nhận thức:} Việc sử dụng biểu đồ vùng cho Open Interest giúp người xem cảm nhận được "độ dày" và sự tích lũy của vị thế trên thị trường, trong khi biểu đồ đường cho giá giúp tập trung vào sự biến động chính xác.
\end{itemize}

\subsubsection{Phân tích thanh lý}
\begin{figure}[H]
    \centering
    \includegraphics[width=\textwidth]{img/eth_liq.png}
    \caption{Biểu đồ Thanh lý Long/Short và Mất cân bằng thanh lý}

\textbf{Giải thích thiết kế:}
\begin{itemize}
    \item \textbf{Dữ liệu:} Giá trị thanh lý và phía thanh lý.
    \item \textbf{Bố cục:} Sử dụng bố cục đa tầng với hai biểu đồ xếp chồng. Biểu đồ trên hiển thị tỷ lệ mất cân bằng dưới dạng đường nét đứt, biểu đồ dưới hiển thị giá trị thanh lý tuyệt đối.
\item \textbf{Lý thuyết màu sắc:} Tuân thủ quy ước chuẩn của ngành tài chính: Màu xanh (\#00e5cc) cho thanh lý vị thế Long và màu đỏ (\#ff3d71) cho thanh lý vị thế Short. Điều này giúp tận dụng nhận thức thị giác ngay lập tức.
    \item \textbf{Tương tác:} Đồng bộ hóa con trỏ chuỗi giữa các biểu đồ giúp so sánh chính xác thời điểm xảy ra thanh lý lớn với biến động giá tương ứng.
\end{itemize}

\subsubsection{Hồ sơ khối lượng}
\begin{figure}[H]
    \centering
    \includegraphics[width=\textwidth]{img/eth_vol_profile.png}
    \caption{Phân bổ khối lượng giao dịch theo mức giá (ETH)}
\end{figure}

\textbf{Giải thích thiết kế:}
\begin{itemize}
    \item \textbf{Dữ liệu:} Khối lượng tích lũy cho từng khoảng giá.
    \item \textbf{Bố cục:} Biểu đồ thanh nằm ngang. Việc xoay ngang biểu đồ này có mục đích thực tiễn: để trục Y khớp với trục Y của biểu đồ nến, giúp người dùng đối chiếu trực quan các vùng tích lũy khối lượng với các mức giá hỗ trợ/kháng cự.
    \item \textbf{Mã hóa thị giác - Vùng giá trị:} Sử dụng màu sắc để phân biệt vùng chiếm 70\% khối lượng (màu tím đậm) và vùng ngoài (màu xám nhạt). Đây là kỹ thuật sử dụng màu sắc để phân loại dữ liệu quan trọng.
    \item \textbf{Dấu hiệu - Điểm kiểm soát:} Một đường kẻ nổi bật đánh dấu mức giá có khối lượng giao dịch lớn nhất, giúp xác định tâm điểm của thị trường.
\end{itemize}

\textbf{Biện luận thiết kế:}
\begin{itemize}
    \item \textbf{Nhiệm vụ:} So sánh sự tương quan giữa nhiều cặp tài sản cùng lúc để hỗ trợ quản trị rủi ro danh mục.
    \item \textbf{Ánh xạ thị giác:} Sử dụng ma trận nhiệt. Trục X và Y đều là danh sách các đồng coin.
    \item \textbf{Kênh - Màu sắc:} Cường độ màu sắc mã hóa giá trị hệ số tương quan từ -1 đến 1. Màu xanh đậm thể hiện tương quan thuận mạnh, màu đỏ thể hiện tương quan nghịch.
    \item \textbf{Nguyên lý nhận thức:} Việc hiển thị trực tiếp con số trên từng ô giúp người dùng kết hợp giữa nhận diện nhanh qua màu sắc và đọc thông tin chính xác khi cần.
\end{itemize}

\subsection{Bố cục và nguyên tắc thị giác tổng quát}
Dashboard được thiết kế theo nguyên tắc: tổng quan trước, phóng to hơn và lọc thông tin, sau đó xem chi tiết cần quan tâm.

\begin{itemize}
    \item \textbf{Bố cục phân tầng:} Các chỉ số tổng hợp quan trọng nhất được đặt ở trên cùng, các biểu đồ nến và thanh lý (chi tiết nhất) được đặt ở trung tâm, và các biểu đồ phân tích tương quan/thống kê ở phía dưới.
    \item \textbf{Nguyên tắc Gestalt:}
    \begin{itemize}
        \item \textbf{Gần gũi:} Các bộ lọc (Selectbox, Date input) được đặt ngay sát biểu đồ tương ứng để người dùng hiểu được phạm vi tác động.
        \item \textbf{Đồng nhất:} Sử dụng bộ màu Bull/Bear thống nhất trên toàn bộ Dashboard giúp giảm tải nhận thức.
    \end{itemize}
    \item \textbf{Lý thuyết màu sắc:} Sử dụng nền tối giúp các màu sắc neon (xanh, đỏ, cam) của biểu đồ nổi bật hơn (tương phản cao), giảm mỏi mắt khi quan sát dữ liệu trong thời gian dài.
\end{itemize}

\subsection{Tổng kết và Biện luận thiết kế tổng thể}

\subsubsection{Bố cục}
Dashboard áp dụng nguyên tắc "Tổng quan trước, phóng to và lọc, sau đó xem chi tiết"
\begin{itemize}
    \item \textbf{Tầng 1:} Biểu đồ Giá/OI và So sánh OI cung cấp cái nhìn vĩ mô về dòng tiền và xu hướng giá dài hạn.
    \item \textbf{Tầng 2:} Candlestick kết hợp Hồ sơ khối lượng và phân tích thanh lý giúp trader hiểu sâu về thị trường.
    \item \textbf{Tầng 3:} Ma trận nhiệt và tương quan động hỗ trợ quản trị rủi ro bằng cách cho thấy mối liên hệ giữa các tài sản.
\end{itemize}

\subsubsection{Visual encoding và nhận thức}
\begin{itemize}
    \item \textbf{Độ chính xác:} Các biến định lượng quan trọng như giá và khối lượng được mã hóa bằng \textit{Vị trí}.
    \item \textbf{Bộ lọc:} Các bộ lọc được đặt sát biểu đồ tương ứng và sử dụng chung một bộ màu Bull/Bear cho toàn bộ Dashboard. Điều này giúp giảm đáng kể "Tải trọng nhận thức", giúp trader ra quyết định nhanh trong môi trường biến động cao.
    \item \textbf{Sự phân bậc thị giác:} Sử dụng kích thước biểu đồ lớn hơn cho các thành phần quan trọng (Nến giá) và màu sắc rực rỡ (accent) cho các tín hiệu thanh lý, OI.
\end{itemize}

\subsubsection{Lý thuyết màu sắc và UI/UX}
\begin{itemize}
    \item \textbf{Màu sắc:} Thiết kế sử dụng nền tối (\textit{Dark Mode}) với các màu neon tương phản cao, giảm chói mắt.
    \item \textbf{Tương tác:} Đồng bộ \textit{cross-hair} và \textit{tooltip} cung cấp dữ liệu chi tiết tại từng điểm mà không làm rối giao diện chung
\end{itemize}

\subsubsection{Nhận diện thị trường và tín hiệu giao dịch}
\begin{itemize}
    \item \textbf{Xác định trạng thái:} Kết hợp Giá và Open Interest (OI) giúp phân biệt xu hướng bền vững (có nhiều hợp đồng) và sự cạn kiệt. Giá tăng kèm OI tăng cho thấy lực mua mạnh, ngược lại giá tăng nhưng OI giảm cảnh báo đảo chiều. 
    
        Ví dụ: với ETH ngày 25/2 20h - 26/2 8h thấy OI tăng và giá cũng tăng, có thể vào lệnh Long ngay 26/2 8h giá vào \$3114.4. Thoát lệnh khi OI tăng cao hơn so với lân cận - lúc 27/2 4h giá \$3220.25. Như vậy trader ăn được chênh lệch \$106.15.
    \item \textbf{Tín hiệu vào/ra (Entry/Exit):} Trader tìm điểm vào tại vùng POC (Hồ sơ khối lượng) hoặc khi các coin có tương quan (lúc 1 coin tăng thì vào lệnh 2 coin, thì coin kia cũng tăng). 

        Ví dụ: 
        - 25 - 26/2 có POC là \$3043, kết hợp việc thấy giá coin tăng trong 1 tuần trước đó. Nếu ta vào lệnh Long ngay tại POC, sau đó tại 27/2 4h thoát sẽ là \$3220.5. Như vậy ăn được chênh lệch \$177.5. Kết hợp tương quan giá trong 1 tuần trước 25/2 ở mức rất cao (> 0.7) cho thấy các coin tăng bền 1 tuần.
\end{itemize}

\subsubsection{Kết luận}
Dashboard đã hoàn thành mục tiêu đề ra: giúp các trader và nhà phân tích không chỉ "thấy" dữ liệu mà còn "hiểu" được bản chất của thị trường, kết hợp trực quan hóa đã tạo ra một công cụ phân tích mạnh mẽ.
